def solution(A):
    differences = []
    sumA = sum(A[:1])
    sumB = sum(A[1:])
    diff = abs(sumA - sumB)
    differences.append(diff)

    for P in range(2, len(A)):
        sumA += A[P-1]
        sumB -= A[P-1]
        diff = abs(sumA - sumB)
        differences.append(diff)
    
    differences = sorted(differences)

    return differences[0]
